#!/bin/sh
IMAGE=$1
set -eux; \
  docker pull "${IMAGE}"; \
  echo "version: '3'
services:
  api:
     image: ${IMAGE}
     ports:
      - '18080:18080'
     volumes:
      - './log/api:/app/log'
     entrypoint:
      - /app/layout_api
     restart: always
  task:
    image: ${IMAGE}
    volumes:
      - './log/task:/app/log'
    entrypoint:
      - /app/layout_task
    restart: always" > docker-compose.yml; \

  docker-compose up -d
